' P4 flow
R   --> P41 : Search medicine\nto purchase
P41 --> D3a : Check wholesaler\nstock availability
P41 --> P42 : Medicine found
P42 --> D6  : Fetch active schemes\nfor this retailer
P42 --> D9  : Fetch frequent\nwholesalers
P42 --> R   : Wholesaler list +\nbest schemes ranked
R   --> P43 : Select wholesaler\nand confirm order
P43 --> D4d : Save purchase order
P43 --> W   : Purchase order\nnotification
W   --> P44 : Approve or deny\n(with reason)
P44 --> D4d : Update order status
P44 --> D6  : Apply custom pricing\nand scheme if any
P44 --> R   : Approval / denial\nnotification
P44 --> P45 : If approved
R   --> P45 : Make payment
P45 --> D5  : Record payment\nand generate invoice
P45 --> D4f : Update B2B\ndelivery tracking
P45 --> W   : Confirm dispatch
P45 --> P46 : On delivery received
P46 --> D3b : Add stock to\nretailer inventory
P46 --> R   : Delivery confirmed\n+ invoice download

' P5 flow
W   --> P51 : Select company\nand medicine
P51 --> D9  : Check wholesaler-\ncompany supplier link
P51 --> CO  : Send supplier\nrequest if not linked
CO  --> P51 : Approve supplier link
P51 --> P52 : Link confirmed
P52 --> D4e : Save bulk order
P52 --> CO  : Bulk order\nnotification
CO  --> P53 : Approve or deny\n(with reason)
P53 --> D4e : Update order status
P53 --> W   : Approval / denial\nnotification
P53 --> P54 : If approved
W   --> P54 : Make payment
P54 --> D5  : Record payment\nand generate invoice
P54 --> D4f : Update B2B\ndelivery tracking
P54 --> CO  : Confirm dispatch
P54 --> P55 : On delivery received
P55 --> D3a : Add stock to\nwholesaler inventory
P55 --> W   : Delivery confirmed\n+ invoice download
